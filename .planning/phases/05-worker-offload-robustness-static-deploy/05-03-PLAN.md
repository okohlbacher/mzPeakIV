    ONMESSAGE HANDLER (module scope, after worker instantiation):
    worker.onmessage = (e: MessageEvent<WorkerResponse>): void => {
      const msg = e.data;
      switch (msg.type) {
        case 'progress':
          useStore.setState({ stage: msg.stage });
          break;
        case 'loadResult': {
          // Spread LoadResult fields into state; store no longer holds reader.
          const r = msg.result as LoadResult;
          useStore.setState({
            manifest: r.manifest,
            fileMeta: r.fileMeta,
            stats: r.stats,
            capabilities: r.capabilities,
            grid: r.grid ?? null,
            tic: r.tic ?? null,
            mixedRepresentationWarning: r.mixedRepresentationWarning,
            stage: 'ready',
            error: null,
            selectedIndex: null,
            selectedSpectrum: null,
          });
          break;
        }
        case 'noImaging': {
          const r = msg.result as NonImagingResult;
          useStore.setState({
            manifest: r.manifest,
            fileMeta: r.fileMeta,
            stats: r.stats,
            capabilities: r.capabilities,
            grid: null,
            tic: null,
            stage: 'no-imaging',
            error: null,
            selectedIndex: null,
            selectedSpectrum: null,
          });
          break;
        }
        case 'renderResult':
          if (msg.requestId !== currentRequestId) break; // stale — discard (Pattern 5)
          useStore.setState({
            ionImage: msg.ionImage ?? null,
            ionImageStats: msg.stats ?? null,
            isRendering: false,
          });
          break;
        case 'spectrumResult':
          useStore.setState({
            selectedIndex: msg.spectrum.index,
            selectedSpectrum: msg.spectrum,
          });
          break;
        case 'error':
          useStore.setState({
            stage: 'error',
            error: { class: msg.class, message: msg.message, findings: msg.findings },
            isRendering: false, // CRITICAL: clear flag on error (Pitfall 7)
          });
          break;
      }
    };

    REWRITE ACTION BODIES:

    openUrl(url: string): async no longer needed — use a sync function body:
      openUrl(url: string) {
        set({ ...initialState, stage: 'zip-index' });
        worker.postMessage({ type: 'loadUrl', url } satisfies WorkerRequest);
      }

    openFile(file: File): use sync body with one await for arrayBuffer():
      async openFile(file: File) {
        set({ ...initialState, stage: 'zip-index' });
        const buffer = await file.arrayBuffer();
        worker.postMessage({ type: 'loadFile', bytes: buffer, name: file.name } satisfies WorkerRequest, [buffer]);
      }
    Note: [buffer] in the transfer list — transfers ownership to Worker (Pattern 4).

    selectSpectrum(index: number): route to Worker (Worker holds activeReader):
      selectSpectrum(index: number) {
        worker.postMessage({ type: 'selectSpectrum', index } satisfies WorkerRequest);
        set({ selectedIndex: index }); // optimistic update for UI — actual spectrum arrives via 'spectrumResult' message
      }

    renderIonImage(mz: number, tolDa: number): dispatch to Worker:
      renderIonImage(mz: number, tolDa: number) {
        const { grid, stats } = get();
        if (!grid || !stats) return;
        // Defense-in-depth validation — Worker also validates (T-05-02).
        if (!Number.isFinite(mz) || mz <= 0 || !Number.isFinite(tolDa) || tolDa <= 0) return;
        if (mz - tolDa < 0) return;
        const rid = ++currentRequestId;
        set({ isRendering: true, mzWindow: { mz, tolDa } }); // optimistic mzWindow for spectrum band (SPEC-02)
        worker.postMessage({ type: 'renderIonImage', mz, tolDa, requestId: rid } satisfies WorkerRequest);
      }
    Note: mzWindow is set optimistically here; the ion image itself arrives via 'renderResult'. Remove the explicit mzWindow set from the old renderIonImage try block.

The Zustand pattern for calling setState from outside create() is useStore.setState({...}) — this works because Zustand exposes setState on the store object.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Worker → Main thread | WorkerResponse messages drive all store state; main thread has no Reader access |
| Main thread → Worker | WorkerRequest messages — validated before dispatch (defense-in-depth) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-05 | Tampering | stale renderResult applied to wrong state | mitigate | requestId generation counter; stale responses silently discarded in onmessage |
| T-05-06 | Denial of Service | isRendering stuck true after Worker error | mitigate | error case in onmessage explicitly sets isRendering: false (Pitfall 7) |
| T-05-07 | Tampering | openFile ArrayBuffer transferred twice | mitigate | buffer passed by reference then transferred [buffer]; file.arrayBuffer() is called once; no double-use |
| T-05-SC | Tampering | npm/pip/cargo installs | accept | No new packages; no install tasks in this plan |
</threat_model>
