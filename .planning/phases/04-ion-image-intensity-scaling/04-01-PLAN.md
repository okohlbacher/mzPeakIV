From src/compute/tic.ts (clone template for buildIonImage):
```typescript
// Local structural interfaces — no vendor import, stays above reader boundary
interface XicPointLike {
  index: bigint | number;
  dataArrays: Record<string, ArrayLike<number> | ArrayLike<string> | undefined>;
}
interface XicLike {
  points: XicPointLike[];
}
const INTENSITY_KEY = "intensity array";

export function buildTic(xic: XicLike, grid: ImagingGrid): Float32Array {
  const tic = new Float32Array(grid.width * grid.height);
  const idxToKey = new Map<number, number>();
  for (const [key, sIdx] of grid.coordToSpectrumIndex) {
    idxToKey.set(sIdx, key);           // reverse map: spectrumIndex → grid key
  }
  for (const point of xic.points) {
    const sIdx = Number(point.index);  // FIRST: bigint → number (Pitfall 1)
    const key = idxToKey.get(sIdx);
    if (key === undefined) continue;
    const arr = point.dataArrays[INTENSITY_KEY];
    if (!arr) continue;
    let sum = 0;
    for (let i = 0; i < arr.length; i++) {
      const v = arr[i];
      sum += typeof v === "number" && Number.isFinite(v) ? v : 0;
    }
    tic[key] = sum;
  }
  return tic;
}
```
